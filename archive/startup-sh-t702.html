Example ~/.xinitrc.

Code: Select all

#!/bin/sh

conky &
(sleep 5 wbar -above-desk) &
numlockx &
unclutter &
case $1 in
icewm)
    exec icewm
    ;;

fluxbox)
    exec startfluxbox
    ;;
*)
    exec startfluxbox
    ;;
esac
