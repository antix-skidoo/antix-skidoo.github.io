Code: Select all

#!/bin/bash
page=http://webserver/gnump3d:8888
int=2

while true; do

  sleep $int
  running=$(ps -ef | grep links2 | wc -l)
  echo"running = $running"

  if (("$running" <"2" )); then
    links2 -g $page &
  fi
done
