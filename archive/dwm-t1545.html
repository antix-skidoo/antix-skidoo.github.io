regular dillo.
save this as ~/.dillo/dillorc

Code: Select all

geometry=963x701
use_dicache=YES
vw_fontname=Bitstream Vera Sans Mono
fw_fontname=Bitstream Vera Sans Mono
font_factor=0.8
use_oblique=YES
show_tooltip=YES
limit_text_width=NO
w3c_plus_heuristics=YES
home="http://www.dillo.org/"
search_url="http://www.google.de/linux?num=100&hl=de&q=%s"
bg_color=3f3f3f
text_color=f0dfaf
link_color=7f9f7f
force_my_colors=YES
contrast_visited_color=YES
panel_size=tiny
small_icons=YES
show_back=NO
show_forw=NO
show_home=NO   
show_reload=NO
show_save=NO
show_stop=NO
show_bookmarks=NO
show_search=NO
show_progress_box=NO
fullwindow_start=NO
transient_dialogs=NO
enterpress_forces_submit=NO
generate_submit=NO
